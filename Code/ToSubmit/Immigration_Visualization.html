In [537]:
total_imm_stats = pd.DataFrame(
    np.array(apprehensions['Total'])+ np.array(deported['Total Deportations']) + np.array(lawful['Total'][1:]) + np.array(refugees['Total'])
)

total_imm_stats.columns = ['Total Events']

total_imm_stats['%Apprehensions'] = 100 * apprehensions['Total'] / total_imm_stats['Total Events']
total_imm_stats['%Deporations'] = 100 * deported['Total Deportations'] / total_imm_stats['Total Events']
total_imm_stats['%Lawful'] = 100 * lawful[1:].reset_index()['Total'] / total_imm_stats['Total Events']
total_imm_stats['%Refugees'] = 100 * refugees['Total'] / total_imm_stats['Total Events']
total_imm_stats['%Total'] = 100 * total_imm_stats['%Apprehensions'] + total_imm_stats['%Deporations'] + total_imm_stats['%Lawful'] + total_imm_stats['%Refugees']

total_imm_stats = total_imm_stats.set_index(pd.Index(range(2016, 2002, -1)))
